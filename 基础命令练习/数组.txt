#!/bin/bash

#创建数组
array_one=('d' 'i' 'c' 'o' 'n')

array_two=()
array_two[0]='d'
array_two[1]='i'
array_two[2]='c'
array_two[3]='o'
array_two[4]='n'

echo ${array_one[@]}
echo ${array_two[@]}

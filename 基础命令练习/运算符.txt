#!/bin/bash

:<<!
原生bash不支持简单的数学运算，但是可以通过其他命令来实现，例如awk
和expr，expr最常用，expr是一款表达式计算工具，使用它能完成表达式的
求值操作。
!

#!/bin/bash

val =`expr 2+2`
echo "两数之和为：$val"

-------------------------------------------------------

#!/bin/bash

a=10
b=20

val=`expr $a + $b`
echo "a + b : $val"

#!/bin/bash

a=10
b=20

val=`expr $a + $b`
echo "a + b : $val"

val=`expr $a \* $b`
echo "a * b : $val"

val=`expr $a / $b`
echo "a / b : $val"

if [ $a == $b ]
then
	echo "a==b"
fi
if [ $a != $b ]
then
	echo "a!=b"
fi


-------------------------------------------------------

:<<!
-eq 检测两个数是否相等，相等返回true。 
-ne 检测两个数是否不相等，不想等返回true。
-gt 检测左边的数是否大于右边的，如果是，则返回true。
-lt 检测左边的数是否小于右边的，如果是，则返回true。
-ge 检测左边的数是否大于等于右边的，如果是，则返回true。
-le 检测左边的数是否小于等于右边的，如果是，则返回true。
!
